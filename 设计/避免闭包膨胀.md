

如果不使用循环，而是直接写死，会制造1000条闭包，而且会生成1000个内存实例结构。 导致编译时间巨长，体积碰撞。在编写程序时应该把大量的的完全相同的控件使用循环来创建，而不是复制多次来实现。

```rust
fn main() {
    // 创建主窗口
    let window = Window::builder()
        .wait_v_sync(false)
        .title("音乐播放器")
        .size(800, 600)
        .view(move |_| {
            // // 构建垂直布局的UI
            v_stack()
                // .apply_theme(&theme)
                // 音量显示标签
                .views({
                    let mut images = vec![];
                    for _ in 0..1000 {
                        images.push(view!(image(|| C.image.get())))
                    }
                    let mut views = views![
                        label(|| format!("音量: {:.1}", C.ui.volume.get())),
                        // 播放按钮
                        button(|| "播放")
                            .style(|s| s.normal().border(Border {
                                border_size: Rect {
                                    left: Length(1.),
                                    right: Length(1.),
                                    top: Length(1.),
                                    bottom: Length(1.),
                                },
                                background: Default::default(),
                            }))
                            .layout(|l| l.normal().align_self(AlignSelf::Start))
                            .on_click(move || {
                                C.audio_core.lock().play();
                            })
                            .focus_index(|| 0),
                        // 暂停按钮
                        button(|| "暂停").on_click(|| {
                            C.audio_core.lock().pause();
                        }),
                        // image(include_bytes!("../../../res/Nav1.png").to_vec()),
                        // image_test(),
                        // text_test(),
                        // draw_quad_test(),
                        // svg_test(),
                        // fill_quad_test(),
                        // path_stroke_test(),
                        // path_fill_test(),
                        // blur_region_test(),
                    ];
                    views.extend(images);
                    views
                })
                .layout(|s| s.align_items(AlignItems::Center))
        })
        .build();
}
```
